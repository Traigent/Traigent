Chunking Strategy

def _chunk_text(text, chunk_size, overlap):
words = text.split()
chunks = []
i = 0
while i < len(words):
    chunks.append(" ".join(words[i:i + chunk_size]))
    i += max(1, chunk_size - overlap)
return chunks
